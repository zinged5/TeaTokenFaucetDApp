    <button onclick="requestTokens()">Request Tokens</button>


    <script>

        const contractAddress = "<TOKEN_FAUCET_CONTRACT_ADDRESS>";

        const abi = [...]; // Replace with the actual ABI of your TokenFaucet contract

        const web3 = new Web3(new Web3.providers.HttpProvider("https://rpc.testnet.tea.network"));


        const faucetContract = new web3.eth.Contract(abi, contractAddress);


        async function requestTokens() {

            const accounts = await web3.eth.requestAccounts();

            const userAddress = accounts[0];


            try {

                await faucetContract.methods.requestTokens(100).send({ from: userAddress });

                alert("Tokens requested successfully!");

            } catch (error) {

                console.error("Error requesting tokens:", error);

            }

        }

    </script>
